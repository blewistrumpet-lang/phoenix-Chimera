cmake_minimum_required(VERSION 3.15)
project(PlatformCompatibilityTest)

set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED YES)

# JUCE setup
add_subdirectory(/Users/Branden/JUCE JUCE)

# Add the test executable
add_executable(test_platform_compatibility
    test_platform_compatibility.cpp
    ../JUCE_Plugin/Source/EngineFactory.cpp
    ../JUCE_Plugin/Source/ParametricEQ.cpp
)

    ../JUCE_Plugin/Source
    ../JUCE_Plugin/JuceLibraryCode
)

    juce::juce_audio_basics
    juce::juce_core
    juce::juce_dsp
)

target_compile_definitions(test_platform_compatibility PRIVATE
    JUCE_STANDALONE_APPLICATION=1
    JUCE_USE_CURL=0
    JUCE_WEB_BROWSER=0
)
