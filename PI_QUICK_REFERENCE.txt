1️⃣  TRANSFER TO PI (Choose one method)
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

Network (fastest):
------------------
rsync -avz --progress pi_deployment/ pi@raspberrypi.local:~/ChimeraPhoenix/

USB Drive:
----------
cp -r pi_deployment /Volumes/YOUR_DRIVE/ChimeraPhoenix
# Then on Pi:
cp -r /media/pi/YOUR_DRIVE/ChimeraPhoenix ~/


━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
2️⃣  BUILD ON PI (Run these commands on Pi)
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

ssh pi@raspberrypi.local

cd ~/ChimeraPhoenix
chmod +x pi_setup.sh
./pi_setup.sh

⏱  Build time: 10-20 minutes
💾 Will install: build tools, JUCE, JACK audio
🔨 Compiles with ARM optimizations


━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
3️⃣  RUN CHIMERA PHOENIX
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

~/chimera_run.sh

Or manually:
------------
# Start JACK audio
jackd -R -dalsa -dhw:1 -r48000 -p512 -n3 &

# Launch plugin
cd ~/ChimeraPhoenix/JUCE_Plugin/Builds/LinuxMakefile/build
./ChimeraPhoenix

🔧 TROUBLESHOOTING
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

Build fails:
sudo apt update && sudo apt install -y build-essential libasound2-dev

No sound:
aplay -l  # List audio devices
# Then restart JACK with correct device number

Trinity server not connecting:
# On Mac, ensure server is running:
cd AI_Server
python3 server_trinity_complete.py

# Test connection from Pi:
curl http://YOUR_MAC_IP:8000/health
